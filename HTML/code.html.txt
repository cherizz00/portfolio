<body>
  <header>
    <h1>Mini Shopify Storefront</h1>
  </header>
  <div class="container">
    <div class="products" id="products"></div>
    <div class="cart" id="cart">
      <h2>Shopping Cart</h2>
      <div id="cart-items"></div>
      <div class="cart-total" id="cart-total"></div>
    </div>
  </div>
  <script>
    // Demo products
    const PRODUCTS = [
      { id: 1, name: "T-shirt", price: 499, img: "https://img.icons8.com/color/96/000000/t-shirt.png" },
      { id: 2, name: "Sneakers", price: 2999, img: "https://img.icons8.com/color/96/000000/sneakers.png" },
      { id: 3, name: "Backpack", price: 999, img: "https://img.icons8.com/color/96/000000/backpack.png" },
      { id: 4, name: "Watch", price: 1899, img: "https://img.icons8.com/color/96/000000/watch.png" }
    ];
    const cart = {};

    // Render products
    function renderProducts() {
      const el = document.getElementById("products");
      el.innerHTML = '';
      PRODUCTS.forEach(product => {
        el.innerHTML += `
        <div class="product">
          <img src="${product.img}" alt="${product.name}">
          <h3>${product.name}</h3>
          <p>₹${product.price}</p>
          <button onclick="addToCart(${product.id})">Add to Cart</button>
        </div>
        `;
      });
    }

    // Cart functionality
    function addToCart(id) {
      cart[id] = (cart[id] || 0) + 1;
      renderCart();
    }

    function removeFromCart(id) {
      if (cart[id] > 1) cart[id]--;
      else delete cart[id];
      renderCart();
    }

    function renderCart() {
      const itemsEl = document.getElementById("cart-items");
      const totalEl = document.getElementById("cart-total");
      itemsEl.innerHTML = '';
      let total = 0;
      Object.keys(cart).forEach(id => {
        const prod = PRODUCTS.find(p => p.id == id);
        const qty = cart[id];
        total += prod.price * qty;
        itemsEl.innerHTML += `
          <div class="cart-item">
            <span>${prod.name} x${qty}</span>
            <span>₹${prod.price * qty}</span>
            <button onclick="removeFromCart(${id})">-</button>
          </div>
        `;
      });
      totalEl.innerText = "Total: ₹" + total;
      if (!Object.keys(cart).length) itemsEl.innerHTML = "<i>Your cart is empty.</i>";
    }

    // Initial render
    renderProducts();
    renderCart();

    // Make functions global
    window.addToCart = addToCart;
    window.removeFromCart = removeFromCart;
  </script>
